PUSH lmain@@Init

simple_heap_init: 
RET 0 ; Retorna inmediatamente

simple_malloc:
LOADFP	; Inicialización unidad
LOADSP
STOREFP ; Finaliza inicialización del RA
LOADHL	; hl
DUP	; hl
PUSH 1	; 1
ADD	; hl+1
STORE 4 ; Guarda el resultado (un puntero a la primer celda de la región de memoria)
LOAD 3	; Carga la cantidad de celdas a alojar (parámetro que debe ser positivo)
ADD
STOREHL ; Mueve el heap limit (hl). Expande el heap
STOREFP
RET 1	; Retorna eliminando el parámetro



.DATA

.CODE
lmain@@Init:
LOADFP  ; Guarda en la pila el enlace dinámico al comienzo del RA del llamador.
LOADSP  ; Apila el lugar donde comienza el RA de la unidad llamada
STOREFP ; Actualiza el FP para que apunte al comienzo del RA de la unidad llamada.
RMEM 1 ; Reservamos un lugar para declaracion de variable local
PUSH 1 ; Apilo true
STORE 0 ; almaceno el valor de la expresion en la variable local
RMEM 1 ; Reservamos un lugar para declaracion de variable local
PUSH 'e' ; Apilo un literal caracter
STORE -1 ; almaceno el valor de la expresion en la variable local
RMEM 1 ; Reservamos un lugar para declaracion de variable local
PUSH 9 ; Apilo un literal entero
STORE -2 ; almaceno el valor de la expresion en la variable local
PUSH 9 ; Apilo un literal entero
PUSH ldebugPrint@int@Object
CALL
LOAD 0 ; Apilo el valor de la variable local o parametro
PUSH lprintB@boolean@System
CALL
PUSH 0 ; Apilo false
PUSH lprintB@boolean@System
CALL
PUSH 'c' ; Apilo un literal caracter
PUSH lprintC@char@System
CALL
LOAD -2 ; Apilo el valor de la variable local o parametro
PUSH lprintI@int@System
CALL
PUSH lprintln@@System
CALL
PUSH 1 ; Apilo true
PUSH lprintBln@boolean@System
CALL
LOAD -1 ; Apilo el valor de la variable local o parametro
PUSH lprintCln@char@System
CALL
PUSH 93 ; Apilo un literal entero
PUSH lprintIln@int@System
CALL
FMEM 3 ; Liberamos las variables locales utilizadas
STOREFP
RET 0 ; Libera los n parametros de la pila

lInit@@:
NOP
